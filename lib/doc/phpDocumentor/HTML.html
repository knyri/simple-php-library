		static method echo_header (line 29)
	
 
	
	

Echos the HEAD tag with the given subtags.

	
    
				
  • access: public
    • 
			

	
	

	    static
		void
		
			echo_header
		
					([string $title = ''], [ $meta = null], [ $link = null], [string|array $base = null], [ $style = null], [ $script = null], [ $ie = null], [string $raw = null])
			

	
			
    
					
  • 
				string
				$title: The content of the TITLE tag.			
    • 
					
  • 
				array
				$meta: An array of array('attrib'=>'value',...)			
    • 
					
  • 
				array
				$link: An array of array('attrib'=>'value',...). Common attribute value pairs are      rel=>stylesheet,href=>url,media=>media type			
    • 
					
  • 
				string|array
				$base: If a string the will set the base href.      If an array it should be an array of ('attrib'=>'value',...)			
    • 
					
  • 
				array
				$style: An array of array('attrib'=>'value',...). A special index 'content' can be set to specify the content between the STYLE tags			
    • 
					
  • 
				array
				$script: An array of array('attrib'=>'value',...). A special index 'content' can be set to specify the content between the SCRIPT tags			
    • 
					
  • 
				array
				$ie: An array of array('lt'|'gt'|'eq'|'lte'|'gte'|'neq',version,link,style,script) where link,style, and script have the same format as $link,$style, and $script.      All may be null but must be set.			
    • 
					
  • 
				string
				$raw: Raw output appended to the very end.			
    •
